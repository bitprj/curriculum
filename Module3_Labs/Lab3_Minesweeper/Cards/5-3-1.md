```python
def mine_repr(self, row_id, col_id):
    cell = self[row_id][col_id]
    if cell.is_visible:
        if cell.is_mine:
            return "M"
        else:
            surr = self.count_surrounding(row_id, col_id)
            return str(surr) if surr else " "
    elif cell.is_flagged:
        return "F"
    else:
        return "X"
```
