```python
def count_surrounding(self, row_id, col_id):
    sum = 0
	for (surr_row, surr_col) in self.get_neighbors(row_id, col_id):
        if (self.is_in_range(surr_row, surr_col) and self[surr_row][surr_col].is_mine): 
        sum += 1
	return sum
```
